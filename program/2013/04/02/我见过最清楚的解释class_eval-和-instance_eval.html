忘了，看一次就能回忆起来 

class A
  # defs here go to A
  puts self  # => A
  class << self
     #defs here go to A's eigenclass
  end
end



A.class_eval do
  #defs here go to A
end



A.instance_eval do
  #defs here go to A's eigenclass   

end



s = "Hello World"



class << s
  #defs here go to s's eigenclass
end
