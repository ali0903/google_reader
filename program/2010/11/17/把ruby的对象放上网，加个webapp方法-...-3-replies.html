运行代码

require 'rubygems'
require 'rack'

class Object
def webapp
class << self
define_method :call do |env|
func, *attrs = env['PATH_INFO'].split('/').reject(&:empty?)
[200, {}, send(func, *attrs)]
end
end
self
end
end

Rack::Handler::Mongrel.run [].webapp, :Port => 9292
